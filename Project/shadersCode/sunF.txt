#version 330 core
out vec4 FragColor;
precision mediump float; 
in vec3 v_frag_coord;
in vec3 v_normal;  
in vec2 v_tex; 
uniform vec3 u_view_pos;
uniform sampler2D u_texture; 
void main() { 
vec3 N = normalize(v_normal);
vec3 V = normalize(u_view_pos - v_frag_coord);
FragColor = texture(u_texture, v_tex); 
}
